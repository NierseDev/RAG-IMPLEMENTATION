Usage Guide

Initialization

Create and configure an API client instance:

// Create client with default configuration
const client = new APIClient();

// Or with custom base URL and config
const client = new APIClient('http://localhost:8000', {
    timeout: 30000,
    retries: 3,
    retryDelay: 1000
});

// Set authentication token
client.setAuthToken('your-token-here');

// Set session ID
client.setSessionId(sessionId);

Error Handling

Handle errors gracefully with the APIError class:

try {
    const result = await client.query('Your question');
    console.log(result);
} catch (error) {
    if (error instanceof APIError) {
        if (error.isTimeout()) {
            console.error('Request timed out');
        } else if (error.isNetworkError()) {
            console.error('Network error');
        } else if (error.isClientError()) {
            console.error('Client error:', error.status);
        } else if (error.isServerError()) {
            console.error('Server error:', error.status);
        }
        console.error('Details:', error.toJSON());
    }
}

Interceptors

Add request/response interceptors for middleware:

// Add request interceptor
client.addRequestInterceptor(async (config) => {
    config.headers['X-Custom-Header'] = 'value';
    return config;
});

// Add response interceptor
client.addResponseInterceptor(async (response) => {
    console.log('Response received:', response);
    return response;
});

// Add error interceptor
client.addErrorInterceptor(async (error) => {
    console.error('Error caught:', error);
    // Return handled response or null to rethrow
    return null;
});

Batch Operations

Execute multiple API calls in parallel or sequence:

// Parallel execution (default)
const results = await client.batch([
    { method: 'get', endpoint: '/health' },
    { method: 'get', endpoint: '/stats' },
    { method: 'post', endpoint: '/query/agentic', body: { query: 'Your question' } }
]);

// Sequential execution
const results = await client.batch(
    [
        { method: 'post', endpoint: '/query/agentic', body: { query: 'Q1' } },
        { method: 'post', endpoint: '/query/agentic', body: { query: 'Q2' } }
    ],
    { parallel: false }
);
